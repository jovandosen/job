//////////////////////////////////////////////////////////////////////////////////
1. List all users

awk -F: '{ print $1}' /etc/passwd
//////////////////////////////////////////////////////////////////////////////////
2. Create new directory and set permissions

mkdir -m 700 test
//////////////////////////////////////////////////////////////////////////////////
3. Show current working directory

pwd

-> command will print something simular to /home/jovan/dota
//////////////////////////////////////////////////////////////////////////////////
4. List all content from some directory, with permissions

ls -la
//////////////////////////////////////////////////////////////////////////////////
5. Check the network connectivity between host and server/host

ping localhost
//////////////////////////////////////////////////////////////////////////////////
6. Move through directories

cd
//////////////////////////////////////////////////////////////////////////////////
7. Find words from some file by regular expression

grep -w "test" foo.txt
//////////////////////////////////////////////////////////////////////////////////
8. Copy file

cp foo.txt foo-copy.txt
//////////////////////////////////////////////////////////////////////////////////
9. Move file

mv foo2.txt dd/
//////////////////////////////////////////////////////////////////////////////////
10. Remove file

rm test.txt 
//////////////////////////////////////////////////////////////////////////////////
11. Create file 

touch bar.txt
//////////////////////////////////////////////////////////////////////////////////
12. Remove empty directory

rmdir cc/
//////////////////////////////////////////////////////////////////////////////////
13. Copy directory and its content

cp -R bb/ bb-copy
//////////////////////////////////////////////////////////////////////////////////
14. Remove directory and its content

rm -R bb-copy/
//////////////////////////////////////////////////////////////////////////////////
15. Update versions of all packages

sudo apt update 
sudo apt-get update
//////////////////////////////////////////////////////////////////////////////////
16. Upgrade packages - get the latest packages changes

sudo apt upgrade
sudo apt-get upgrade
//////////////////////////////////////////////////////////////////////////////////
17. Show list of all upgradable packages

sudo apt list --upgradable
//////////////////////////////////////////////////////////////////////////////////
18. Add new user

sudo adduser username
//////////////////////////////////////////////////////////////////////////////////
19. Remove user

sudo deluser username
//////////////////////////////////////////////////////////////////////////////////
20. Change owner of files or directories

sudo chown user:group dirname
chown -R USER:GROUP DIRECTORY
chown -R www-data: /var/www

-> -R flag is used when we want to apply changes to all subdirectories

sudo chown user:group filename
