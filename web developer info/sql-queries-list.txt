ALTER TABLE users AUTO_INCREMENT = 1;

SELECT * FROM users
LIMIT 10
OFFSET 0;

//////////////////////////////////////////////////////////////////////////////////////////////

// pronadji sve korisnike cije ime pocinje sa j //

SELECT * FROM users WHERE name LIKE 'j%';

//////////////////////////////////////////////////////////////////////////////////////////////

// pronadji sve korisnike cije ime sadrzi slovo j (bilo gde) //

SELECT * FROM users WHERE name LIKE '%j%';

//////////////////////////////////////////////////////////////////////////////////////////////

// pronadji sve korisnike cije ime, email ili role-a sadrzi slovo j (bilo gde) //

SELECT * FROM users WHERE name LIKE '%j%' OR email LIKE '%j%' OR role LIKE '%j%';
